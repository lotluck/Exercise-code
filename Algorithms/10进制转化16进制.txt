#include<stdio.h>

int main()  
{  
    long long value;
	int num=0,i;
	int a[10];
   while(scanf("%I64d",&value)!=EOF){ 	
   	num=0;
   	if(value==0)
   	printf("0\n");
   	else
   	{  
	   	while(value)
	   {
		a[num++]=value%16;
		value/=16;
	   }
	
	for(i=num-1;i>=0;i--)
	{
		switch(a[i])
		{
		  case 10:printf("A");break;
		  case 11:printf("B");break;
		  case 12:printf("C");break;
		  case 13:printf("D");break;
		  case 14:printf("E");break;
		  case 15:printf("F");break;
		  default: printf("%d",a[i]);
		}
		
	}
	printf("\n");
   	  	
   }
   		  		
   	}
   
    
    return 0;  
}
