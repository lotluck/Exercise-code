#include<cstdio>
#include<cstring> 
#include<cmath>
#include<cctype>
int main()
{
	char s[10010];
	int i, n, t = 0;
	scanf("%s", s);
	n = strlen(s);
	for(i = 0; i < n; i++)
	{
		if(isdigit(s[i]))
			t = t + (s[i]-'0')*pow(16, n-1-i);
		else t = t + (s[i]-'A'+10)*pow(16, n-1-i);
	}
	printf("%d\n", t);
	return 0;
}
