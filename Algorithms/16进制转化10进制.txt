#include<stdio.h>
#include<string.h>
#include<math.h>
int main()  
{  
    char a[8];
    int i=0,length=0;
    long int value=0;
    gets(a);    
    length=strlen(a);
    while(length)
    {
        if(a[length-1]>='A'&&a[length-1]<='F')
           value+=(a[length-1]-'7')*pow(16,i++);
        else
           value+=(a[length-1]-'0')*pow(16,i++);
           
        length--;
    	
    }
    printf("%I64d",value);
   
    
    return 0;  
}
