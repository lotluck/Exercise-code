#include <stdio.h>
#include <string.h>

#define MAXLEN 100
#define min(a,b,c) (((a<b?a:b)<c)?(a<b?a:b):(c))


int c[MAXLEN][MAXLEN] = {0};

void cls_length(char *sa,char *sb,int m,int n)
{
    int i = 0;
    int j = 0;
    int is_equal = 0;

    for(i = 0; i <= m; i++){
        c[i][0] = i;
    }//end for
    for(j = 0; j <= n; j++){
        c[0][j] = j;
    }//end for

    for(i = 1; i <= m; i++){
        for(j = 1; j <= n; j++){
                 is_equal = (sa[i-1] == sb[j-1])?0:1;
                 c[i][j] = min(c[i-1][j]+1,c[i][j-1]+1,
                               c[i-1][j-1] + is_equal);
        }//end for
    }//end for
}


void print_matrix_cls(int m,int n)
{
    int i = 0;
    int j = 0;

    for(i = 0; i <= m; i++){
        for(j = 0; j <= n; j++){
            printf("%d ",c[i][j]);
        }//end for 2
        printf("\n");
    }//end for 1
    printf("\n\n\nmin edit is %d\n\n\n",c[m][n]);
}

static int m[MAXN][MAXN];

void MaxDp(int *a,int n,int k)
{
    int i = 0;
    int j = 0;
    int d = 0;
    int max = 0;
    int temp = 0;

    for(i = 1; i <= n; i++){
        m[i][1] = w[1][i];
    }//end for

    for(i = 1; i <= n; i++){
        for(j = 2; j <= k; j++){
            max = 0;
            for(d = 1; d < i; d++){
				temp = m[d][j-1]*w[d+1][i];
                if(temp > max){
                    max = temp;
                }//end if
                m[i][j] = max;
            }//end for 3
        }//end for 2
    }//end for 1
}

void init_matrix(int *a,int n)
{
	int i = 0;
	int j = 0;

    for(i = 1; i <= n; i++){
        w[i][i] = a[i];
        for(j = i+1; j <= n; j++){
            w[i][j] = w[i][j - 1]*10 + a[j];
        }//end for 2
    }//end for 1

}
